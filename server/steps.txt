24 -> sudo rm /etc/nginx/sites-enabled/default
25 -> sudo nano /etc/nginx/sites-available/blog-app
26 -> server {
    listen 80;

    server_name _;

    location / {
        root /home/ubuntu/Blog1/client/dist;
        index index.html;
        try_files $uri /index.html;
    }

    location /api {
        proxy_pass http://localhost:5000;
    }
}

27-> sudo ln -s /etc/nginx/sites-available/blog-app /etc/nginx/sites-enabled/
